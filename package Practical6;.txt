package Practical6;

import java.net.*;

public class InetAddressDemo {

    public static void main(String[] args) {
        try {
            // 1. Get local host
            InetAddress local = InetAddress.getLocalHost();
            System.out.println("Local Host:");
            System.out.println("Host Name: " + local.getHostName());
            System.out.println("IP Address: " + local.getHostAddress());

            // 2. Get by name (example: google.com)
            InetAddress google = InetAddress.getByName("www.google.com");
            System.out.println("\nGoogle Host:");
            System.out.println("Host Name: " + google.getHostName());
            System.out.println("IP Address: " + google.getHostAddress());

            // 3. Get all addresses
            InetAddress[] addresses = InetAddress.getAllByName("www.google.com");
            System.out.println("\nAll IP Addresses of Google:");
            for (InetAddress addr : addresses) {
                System.out.println(addr.getHostAddress());
            }

            // 4. Check loopback
            InetAddress loop = InetAddress.getLoopbackAddress();
            System.out.println("\nLoopback Address: " + loop.getHostAddress());

        } catch (Exception e) {
            System.out.println(e);
        }
    }
}
